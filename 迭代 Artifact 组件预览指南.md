## 🎯 使用示例

### 在 chat-lite 中使用

```typescript
// 显示迭代概览
viewer.content = {
  kind: "iteration-overview",
  data: {
    overview: {
      id: "iter-xxx",
      name: "Sprint 1",
      status: "active",
      stats: { ... }
    }
  }
};

// 显示迭代列表
viewer.content = {
  kind: "iteration-list",
  data: {
    iterations: [...],
    selectedId: "iter-xxx"
  }
};

// 显示迭代工作项
viewer.content = {
  kind: "iteration-workitems",
  data: {
    workitems: [...],
    viewMode: "list"
  }
};
```
