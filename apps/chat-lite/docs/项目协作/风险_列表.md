        // 简单的交互模拟
        function toggleBatchSelect(checkbox) {
            const selectAll = document.getElementById('selectAll');
            selectAll.checked = checkbox.checked;
            toggleAllRows(selectAll);
        }

        function toggleAllRows(checkbox) {
            const rowCheckboxes = document.querySelectorAll('.row-checkbox');
            rowCheckboxes.forEach(cb => cb.checked = checkbox.checked);
            // 同步批量操作复选框
            const batchCheckbox = document.getElementById('batchSelectAll');
            if (batchCheckbox) batchCheckbox.checked = checkbox.checked;
        }

        function updateSelectAll() {
            const rowCheckboxes = document.querySelectorAll('.row-checkbox');
            const selectAll = document.getElementById('selectAll');
            const batchCheckbox = document.getElementById('batchSelectAll');
            const allChecked = Array.from(rowCheckboxes).every(cb => cb.checked);
            selectAll.checked = allChecked;
            if (batchCheckbox) batchCheckbox.checked = allChecked;
        }

        function switchView(mode) {
            const icons = document.querySelectorAll('.view-icon');
            icons.forEach(icon => icon.classList.remove('active'));
            event.currentTarget.classList.add('active');
            alert(`切换到 ${mode} 视图 (演示)`);
        }

        function changeRowsPerPage(select) {

### 设计说明
- **标题与批量操作**：左上角“全部风险”图标使用橙色主题，右侧加入“批量操作”复选框，与图片描述一致。

- **新建按钮**：橙色按钮（风险主题色），点击弹出提示。
- **表格列**：严格按照图片包含编号、标题、状态、优先级、负责人、创建者、操作，并添加了第一列复选框用于批量选择。
- **数据行**：仅显示图片中给出的示例数据：编号`DEMO-35`，标题“风险标题”，状态“待处理”，优先级“中”，负责人邮箱完整，创建者邮箱截断为`lenny540101@163.c`，操作栏有编辑和删除图标。
- **底部信息**：显示“共1条”和“每页显示：25”的下拉选择，以及分页控制（第一页禁用上一页/下一页）。
