            let toast = document.getElementById('toast');
            toast.textContent = msg;
            toast.classList.add('show');
            clearTimeout(toast._timer);
            toast._timer = setTimeout(() => toast.classList.remove('show'), 2000);

        // 标签页切换 (仅移动端有效，PC上标签头隐藏，所以不会触发)
        function switchTab(event, tabId) {
            const header = event.target.closest('.tab-header');
            if (!header) return;
            const section = event.target.closest('.tab-section');
            const btns = header.querySelectorAll('.tab-btn');
            const panes = section.querySelectorAll('.tab-pane');

            btns.forEach(btn => btn.classList.remove('active'));
            panes.forEach(pane => pane.classList.remove('active'));

            event.target.classList.add('active');
            document.getElementById(tabId).classList.add('active');
        }
