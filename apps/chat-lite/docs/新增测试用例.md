        function openDrawer() {

            document.getElementById('drawerOverlay').classList.remove('active');
            document.getElementById('mainDrawer').classList.remove('active');
            document.body.style.overflow = '';
        }

        function insertLink() { let url = prompt('链接URL:','https://'); if(url) document.execCommand('createLink', false, url); }

        function addStep() {
            let tbody = document.getElementById('stepsBody');
            let newRow = tbody.insertRow();
            let rowCount = tbody.rows.length + 1;
            newRow.innerHTML = `<td>${rowCount}</td><td>新步骤</td><td>预期结果</td><td><span class="step-action" onclick="removeStep(this)">✕</span></td>`;
        }
        function removeStep(el) {
            let row = el.closest('tr');
            if (row && document.getElementById('stepsBody').rows.length > 1) {
                row.remove();
                // 重新编号
                let rows = document.getElementById('stepsBody').rows;
                for (let i = 0; i < rows.length; i++) {
                    rows[i].cells[0].innerText = i + 1;
                }
            } else {
                showToast('至少保留一行');
            }
        }

        // ----- 底部弹框通用 -----
        function openSheet(id) {
            if (document.getElementById(id + 'Overlay')) {
                document.getElementById(id + 'Overlay').classList.add('active');
                document.getElementById(id).classList.add('active');
            }
        }
        function closeSheet(id) {
            if (document.getElementById(id + 'Overlay')) {
                document.getElementById(id + 'Overlay').classList.remove('active');
                document.getElementById(id).classList.remove('active');
            }
        }

        // 单选更新
        function selectOption(sheetId, valId, el, value) {
            el.closest('.sheet-body').querySelectorAll('.sheet-option').forEach(o => o.classList.remove('selected'));
            el.classList.add('selected');
            document.getElementById(valId).innerHTML = value;
            document.getElementById(valId).classList.remove('placeholder');
            closeSheet(sheetId);
        }

        // 负责人
        function selectAssignee(el, email, initials) {
            el.closest('.sheet-body').querySelectorAll('.sheet-option').forEach(o => o.classList.remove('selected'));
            el.classList.add('selected');
            document.getElementById('assigneeVal').innerHTML = `<span class="avatar-sm">${initials}</span>${email.split('@')[0]}`;
            closeSheet('assigneeSheet');
        }

        // 优先级
        function selectPriority(el, level, label) {
            el.closest('.sheet-body').querySelectorAll('.sheet-option').forEach(o => o.classList.remove('selected'));
            el.classList.add('selected');
            document.getElementById('priorityVal').innerHTML = `<span class="priority-pill priority-${level}"><span class="priority-dot ph-${level}"></span>${label}</span>`;
            document.getElementById('priorityVal').classList.remove('placeholder');
            closeSheet('prioritySheet');
        }

        function saveMultiSelect(sheetId, valId) {
            let selected = document.querySelectorAll(`#${sheetId} .sheet-option.selected`);
            let valEl = document.getElementById(valId);
            if (selected.length === 0) { valEl.textContent = '标签'; valEl.classList.add('placeholder'); }
            else if (selected.length === 1) { valEl.textContent = selected[0].textContent.replace('✓','').trim(); valEl.classList.remove('placeholder'); }
            else { valEl.textContent = `已选 ${selected.length} 项`; valEl.classList.remove('placeholder'); }
            closeSheet(sheetId);
        }
