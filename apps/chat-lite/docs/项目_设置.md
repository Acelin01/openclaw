<script>
    (function() {
        // 获取所有导航项
        const navItems = document.querySelectorAll('.nav-item');
        // 定义卡片映射 (data-section 的值 对应 卡片ID)
        const sectionMap = {
            'basic': 'section-basic',
            'members': 'section-members',
            'roles': 'section-roles',
            'notify': 'section-notify',
            'review': 'section-review'
        };
        // 获取所有卡片 (settings-section)
        const allSections = document.querySelectorAll('.settings-section');
        const placeholder = document.getElementById('section-placeholder');

        // 去除所有nav-item的active类
        function removeActiveClass() {
            navItems.forEach(item => item.classList.remove('active'));
        }

        // 隐藏所有卡片，并返回
        function hideAllSections() {
            allSections.forEach(section => section.style.display = 'none');
        }

        // 根据sectionId显示对应卡片 (如果不存在则显示占位)
        function showSection(sectionId) {
            hideAllSections();
            if (sectionId && document.getElementById(sectionId)) {
                document.getElementById(sectionId).style.display = 'block';
            } else {
                // 如果映射不到卡片，显示占位卡片
                placeholder.style.display = 'block';
            }
        }

        // 为每个nav-item添加点击事件
        navItems.forEach(item => {
            item.addEventListener('click', function(e) {
                e.preventDefault();   // 防止任何可能的默认行为
                const sectionKey = this.dataset.section; // 获取 data-section 的值

                // 高亮当前项
                removeActiveClass();
                this.classList.add('active');

                // 根据sectionKey查找对应的卡片ID
                if (sectionKey && sectionMap[sectionKey]) {
                    showSection(sectionMap[sectionKey]);
                } else {
                    // 没有映射关系，显示占位卡片
                    showSection(null); // 会触发placeholder
                }
            });
        });

        // 初始状态：确保基本信息高亮且显示 (基本信息已在HTML中有active类，但以防万一)
        // 但可能其他nav-item意外有active，我们强制执行一次同步
        const defaultActive = document.querySelector('.nav-item[data-section="basic"]');
        if (defaultActive) {
            removeActiveClass();
            defaultActive.classList.add('active');
            showSection('section-basic');
        } else {
            // fallback: 显示占位
            showSection(null);
        }
    })();
</script>

### 菜单切换与内容联动
页面核心是左侧导航与右侧内容的动态联动。您可以通过点击左侧菜单来浏览不同的设置模块。

-   **菜单交互**：左侧“项目设置”和“工作项设置”下的所有选项均可点击。点击时，该菜单项会高亮显示。
-   **内容切换**：右侧内容区域会跟随菜单切换。
    -   点击**基本信息、项目成员、角色权限、通知、评审**五项时，会分别显示对应的详细配置卡片（包含表单、表格等原有内容）。
    -   点击其他菜单项（如“项目状态”、“需求”等），则会显示一个统一的占位卡片，提示当前为示例演示。
-   **初始状态**：页面加载后，默认选中“基本信息”菜单，并展示其内容。
