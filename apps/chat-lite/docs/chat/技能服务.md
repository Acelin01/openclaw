  const pills = area.querySelectorAll('.tag-pill');

  const pill = document.createElement('span');
  pill.className = 'tag-pill';
  pill.innerHTML = `${val} <button onclick="removeTag(this,event)">×</button>`;
  area.insertBefore(pill, input);
  input.value = '';
}
function removeTag(btn, e) {
  e.stopPropagation();
  btn.parentElement.remove();
}

// ── Toast ──
let toastCount = 0;
function toast(type, title, sub = '') {
  const icons = { success: '✓', error: '✕', info: 'ℹ' };
  const stack = document.getElementById('toastStack');
  const id = 'toast-' + (++toastCount);
  const el = document.createElement('div');
  el.className = 'toast';
  el.id = id;
  el.innerHTML = `
    <div class="toast-icon ${type}">${icons[type]}</div>
    <div class="toast-body"><strong>${title}</strong>${sub ? `<span>${sub}</span>` : ''}</div>
  `;
  stack.appendChild(el);
  setTimeout(() => { el.style.opacity = '0'; el.style.transform = 'translateX(10px)'; el.style.transition = '0.25s ease'; setTimeout(() => el.remove(), 250); }, 2600);
}
