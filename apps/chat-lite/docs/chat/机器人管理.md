// ── Drawer steps ──
let currentStep = 1;
function setStep(n) {
  for (let i = 1; i <= 4; i++) {
    const ws = document.querySelectorAll('.ws')[i-1];
    const content = document.getElementById(`stepContent${i}`);
    if (ws) {
      ws.classList.remove('active','done');
      if (i < n) ws.classList.add('done');
      else if (i === n) ws.classList.add('active');
    }
    if (content) content.style.display = i === n ? 'block' : 'none';
  }
  currentStep = n;
  const nextBtn = document.getElementById('nextBtn');
  if (nextBtn) nextBtn.textContent = n === 4 ? '完成绑定' : '下一步 →';
}
function nextStep() {
  if (currentStep === 1) {
    const name = document.getElementById('botName')?.value;
    if (!name) { toast('error','请填写机器人名称','名称为必填项'); return; }
  }
  if (currentStep < 4) { setStep(currentStep + 1); }
  else { finishBind(); }
}
function prevStep() {
  if (currentStep > 1) setStep(currentStep - 1);
}
function finishBind() {
  const name = document.getElementById('botName')?.value || 'NewBot';
  closeDrawer();
  toast('success','绑定成功',`${name} 已绑定 · Token已同步 · 技能审核中`);
}

function toast(type, title, sub = '') {
  const icons = { success: '✓', error: '✕', info: 'ℹ', warn: '⚠' };
  const stack = document.getElementById('toastStack');
  const el = document.createElement('div');
  el.className = `toast ${type === 'error' ? 'error' : type === 'warn' ? 'warn' : ''}`;
  el.innerHTML = `
    <div class="toast-ico ${type === 'warn' ? 'warn' : type}">${icons[type] || 'ℹ'}</div>
    <div class="toast-body"><strong>${title}</strong>${sub ? `<span>${sub}</span>` : ''}</div>
  `;
  stack.appendChild(el);
  setTimeout(() => {
    el.style.cssText += 'opacity:0;transform:translateX(10px);transition:0.25s ease';
    setTimeout(() => el.remove(), 260);
  }, 2800);
}
