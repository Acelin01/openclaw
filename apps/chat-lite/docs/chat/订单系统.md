  const s = String(cdSeconds % 60).padStart(2,'0');
  const el = document.getElementById('countdownDisplay');
  if (el) el.textContent = `${h}:${m}:${s}`;
}

function toast(type, title, sub = '') {

  el.className = 'toast';
