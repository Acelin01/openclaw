function switchView(view, sidebarEl) {
  const board = document.getElementById('boardView');
  const gantt = document.getElementById('ganttView');
  const vbBoard = document.getElementById('vbBoard');
  const vbGantt = document.getElementById('vbGantt');

  if (view === 'board') {
    board.style.display = 'flex';
    gantt.style.display = 'none';
    vbBoard.classList.add('active'); vbGantt.classList.remove('active');
  } else {
    board.style.display = 'none';
    gantt.style.display = 'flex';
    vbBoard.classList.remove('active'); vbGantt.classList.add('active');
    renderGantt();
  }

  if (sidebarEl) {
    document.querySelectorAll('.sb-item').forEach(i => {
      if (i.textContent.includes('看板') || i.textContent.includes('甘特')) i.classList.remove('active');
    });
    sidebarEl.classList.add('active');
  }
}
