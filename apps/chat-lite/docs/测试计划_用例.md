<script>
    // 切换二级标签样式
    function switchSecTab(el, tabName) {
        document.querySelectorAll('.sec-tab').forEach(t => t.classList.remove('active'));
        el.classList.add('active');
        // 实际项目中会加载对应数据，此处仅演示
        console.log('切换到：', tabName);
    }

    // 设置chip激活（全部用例 / uxin）
    function setActiveChip(chip) {
        document.querySelectorAll('.chip').forEach(c => c.classList.remove('active'));
        chip.classList.add('active');
        // 如果是uxin，可以模拟过滤，但为了视觉演示不变
    }

    // 全选/取消全选
    function toggleSelectAll(source) {
        const checkboxes = document.querySelectorAll('.case-checkbox');
        checkboxes.forEach(cb => {
            if (cb !== source) cb.checked = source.checked;
            // 简单高亮行 (可选)
            const row = cb.closest('tr');
            if (row) {
                if (source.checked) row.classList.add('selected');
                else row.classList.remove('selected');
            }
        });
    }

    // 单行选择，取消全选状态
    function toggleRowSelect(cb) {
        const row = cb.closest('tr');
        if (cb.checked) {
            row.classList.add('selected');
        } else {
            row.classList.remove('selected');
            document.getElementById('selectAll').checked = false;
        }
        // 检查是否所有都已选，更新全选状态 (简单逻辑)
        const allCheckboxes = document.querySelectorAll('.case-checkbox:not(#selectAll)');
        const allChecked = Array.from(allCheckboxes).every(c => c.checked);
        document.getElementById('selectAll').checked = allChecked;
    }

    // 查看用例详情 (演示)
    function viewCase(id) {
        alert(`打开用例 ${id} 详情`);
    }

    // 搜索过滤（前端模拟）
    function filterCases() {
        const keyword = document.getElementById('searchCases').value.toLowerCase();
        const rows = document.querySelectorAll('#caseTableBody tr');
        let visibleCount = 0;
        rows.forEach(row => {
            const title = row.querySelector('.case-title')?.textContent.toLowerCase() || '';
            const id = row.querySelector('.case-id')?.textContent.toLowerCase() || '';
            if (title.includes(keyword) || id.includes(keyword) || keyword === '') {
                row.style.display = '';
                visibleCount++;
            } else {
                row.style.display = 'none';
            }
        });
        // 更新底部计数（简单）
        document.querySelector('.filter-meta strong').textContent = visibleCount;
        // 全选复选框重置
        document.getElementById('selectAll').checked = false;
    }

    // 初始化计数
    window.onload = function() {
        const total = document.querySelectorAll('#caseTableBody tr').length;
        document.querySelector('.filter-meta strong').textContent = total;
    };
</script>

### 用例管理页面的功能模块
页面围绕测试用例的筛选与操作展开，主要分为以下几个区域：

1.  **状态概览与操作栏**：顶部显示“0%通过率”卡片，下方提供“批量操作”和“添加用例”按钮，右侧是关键字搜索框。
2.  **筛选导航**：通过“用例库/需求”二级标签和“全部用例/uxin”筛选标签，您可以快速切换用例视图。
3.  **用例列表**：表格展示了编号、标题、创建时间、优先级、执行人和状态等核心字段。每行前有复选框，支持批量选择。缺陷列可直接关联到具体缺陷编号。
4.  **底部信息**：再次显示当前选中的筛选条件（全部用例）和分页控件，方便浏览。
