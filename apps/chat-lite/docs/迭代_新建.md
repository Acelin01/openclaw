            // 渲染日历
            function renderCalendar(type) {
                const pickerDate = type === 'start' ? startDatePickerDate : endDatePickerDate;
                const monthYearElement = type === 'start' ? startMonthYear : endMonthYear;
                const daysElement = type === 'start' ? startDateDays : endDateDays;
                const selectedDate = type === 'start' ? selectedStartDate : selectedEndDate;
                
                // 更新月份年份显示
                const monthNames = ['一月', '二月', '三月', '四月', '五月', '六月', 
                                   '七月', '八月', '九月', '十月', '十一月', '十二月'];
                monthYearElement.textContent = `${pickerDate.getFullYear()}年${monthNames[pickerDate.getMonth()]}`;
                
                // 获取当月第一天和最后一天
                const firstDay = new Date(pickerDate.getFullYear(), pickerDate.getMonth(), 1);
                const lastDay = new Date(pickerDate.getFullYear(), pickerDate.getMonth() + 1, 0);
                
                // 获取第一天是星期几（0-星期日，6-星期六）
                const firstDayOfWeek = firstDay.getDay();
                
                // 获取上个月的最后几天
                const prevMonthLastDay = new Date(pickerDate.getFullYear(), pickerDate.getMonth(), 0).getDate();
                
                // 清空日期容器
                daysElement.innerHTML = '';
                
                // 添加上个月的日期
                for (let i = firstDayOfWeek - 1; i >= 0; i--) {
                    const day = document.createElement('div');
                    day.className = 'day other-month';
                    day.textContent = prevMonthLastDay - i;
                    daysElement.appendChild(day);
                }
                
                // 获取今天的日期
                const today = new Date();
                const todayStr = `${today.getFullYear()}-${today.getMonth()}-${today.getDate()}`;
                
                // 添加当月的日期
                for (let i = 1; i <= lastDay.getDate(); i++) {
                    const day = document.createElement('div');
                    day.className = 'day';
                    day.textContent = i;
                    
                    // 检查是否是今天
                    const dayDate = new Date(pickerDate.getFullYear(), pickerDate.getMonth(), i);
                    const dayStr = `${dayDate.getFullYear()}-${dayDate.getMonth()}-${dayDate.getDate()}`;
                    
                    if (dayStr === todayStr) {
                        day.classList.add('today');
                    }
                    
                    // 检查是否是选中的日期
                    if (selectedDate && 
                        selectedDate.getFullYear() === pickerDate.getFullYear() && 
                        selectedDate.getMonth() === pickerDate.getMonth() && 
                        selectedDate.getDate() === i) {
                        day.classList.add('selected');
                    }
                    
                    // 添加点击事件
                    day.addEventListener('click', function() {
                        selectDate(type, i);
                    });
                    
                    daysElement.appendChild(day);
                }
                
                // 添加下个月的日期（补全6行）
                const totalDays = firstDayOfWeek + lastDay.getDate();
                const remainingDays = 42 - totalDays; // 6行 * 7天 = 42个格子
                
                for (let i = 1; i <= remainingDays; i++) {
                    const day = document.createElement('div');
                    day.className = 'day other-month';
                    day.textContent = i;
                    daysElement.appendChild(day);
                }
            }

            function selectDate(type, day) {
                const pickerDate = type === 'start' ? startDatePickerDate : endDatePickerDate;
                const selectedDate = new Date(pickerDate.getFullYear(), pickerDate.getMonth(), day);
                const inputElement = type === 'start' ? startDateInput : endDateInput;
                const container = type === 'start' ? startDateContainer : endDateContainer;
                
                // 更新选中的日期
                if (type === 'start') {
                    selectedStartDate = selectedDate;
                    
                    // 如果结束日期早于开始日期，重置结束日期
                    if (selectedEndDate && selectedEndDate < selectedStartDate) {
                        selectedEndDate = null;
                        endDateInput.value = '';
                        endDateContainer.classList.add('placeholder');
                    }
                } else {
                    selectedEndDate = selectedDate;
                    
                    // 如果开始日期晚于结束日期，重置开始日期
                    if (selectedStartDate && selectedStartDate > selectedEndDate) {
                        selectedStartDate = null;
                        startDateInput.value = '';
                        startDateContainer.classList.add('placeholder');
                    }
                }
                
                // 格式化日期显示
                const year = selectedDate.getFullYear();
                const month = (selectedDate.getMonth() + 1).toString().padStart(2, '0');
                const date = selectedDate.getDate().toString().padStart(2, '0');
                inputElement.value = `${year}-${month}-${date}`;
                
                // 移除占位符样式
                container.classList.remove('placeholder');
                
                // 隐藏日期选择弹框
                const picker = type === 'start' ? startDatePicker : endDatePicker;
                picker.classList.remove('active');
                modalOverlay.classList.remove('active');
